radio的基本使用

# 参数支持

参数名称 参数描述 参数类型 默认值

label 单选框的 value 值 string，num,boolean ''

# 选中的样式

.mv-radio.is-checked {
  .mv-radio__input {
    .mv-radio__inner {
      border-color: #409eff;
      background: #409eff;
      &:after {
        transform: translate(-50%,-50%) scale(1);
      }
    }
  }
  .mv-radio__label {
    color: #409eff;
  }
}

# label 与插槽的处理

  • props 接收
props: {
  label: {
    type: String,
    default: ''
  },
  name: {
    type: String,
    default: ''
  }
}
  • 处理插槽
<span class="mv-radio__label">
  <slot></slot>
  <!-- 如果没有插槽内容，那么label就是内容 -->
  <template v-if="!$slots.default">{{label}}</template>
</span>

# v-model 处理

  • 接收 props 数据
name: {
  type: String,
  default: ''
},
value: {
  type: [String, Boolean, Number],
  default: ''
}
  • 结构
<input class="mv-radio__original" type="radio" :name="name" value="label" v-model="model" />
  • 提供计算属性
computed: { model: { get () { return this.value }, set (value) { this.$emit('input', value) } } },
  • 控制选中样式
<label class="mv-radio" :class="{'is-checked': model === label}"></label>
